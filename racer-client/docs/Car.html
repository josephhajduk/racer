pointCar adjusts the sprite of the car actor so that it appears to be pointing in the direction of the supplied directionVector. @angle_index is to allow for improperly ordered sprite sheets

 
  pointCar: (directionVector) ->
    twopi = Math.PI*2
    pie_size = twopi/@angle_index.length
    angle = Vector.unitX.vectorAngle directionVector
    index = Math.floor ((((angle+pie_size/2)%twopi)+twopi)%twopi)/pie_size
    @actor.setSpriteImage(@angle_index[index])
